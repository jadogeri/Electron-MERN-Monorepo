1. Context API (PostContext.js)

import React, { createContext, useReducer, useContext } from 'react';
import axios from 'axios';

const PostContext = createContext();

const initialState = {
  posts: [],
  loading: false,
  error: null,
};

const postReducer = (state, action) => {
  switch (action.type) {
    case 'FETCH_POSTS_REQUEST':
    case 'ADD_POST_REQUEST':
    case 'UPDATE_POST_REQUEST':
      return { ...state, loading: true, error: null };
    case 'FETCH_POSTS_SUCCESS':
      return { ...state, loading: false, posts: action.payload };
    case 'ADD_POST_SUCCESS':
      return { ...state, loading: false, posts: [...state.posts, action.payload] };
    case 'UPDATE_POST_SUCCESS':
      return {
        ...state,
        loading: false,
        posts: state.posts.map((post) =>
          post.id === action.payload.id ? action.payload : post
        ),
      };
    case 'FETCH_POSTS_FAILURE':
    case 'ADD_POST_FAILURE':
    case 'UPDATE_POST_FAILURE':
      return { ...state, loading: false, error: action.payload };
    default:
      return state;
  }
};

export const PostProvider = ({ children }) => {
  const [state, dispatch] = useReducer(postReducer, initialState);

  const fetchPosts = async () => {
    dispatch({ type: 'FETCH_POSTS_REQUEST' });
    try {
      const response = await axios.get('/api/posts'); // Replace with your API endpoint
      dispatch({ type: 'FETCH_POSTS_SUCCESS', payload: response.data });
    } catch (error) {
      dispatch({ type: 'FETCH_POSTS_FAILURE', payload: error.message });
    }
  };

  const addPost = async (newPost) => {
    dispatch({ type: 'ADD_POST_REQUEST' });
    try {
      const response = await axios.post('/api/posts', newPost); // Replace with your API endpoint
      dispatch({ type: 'ADD_POST_SUCCESS', payload: response.data });
    } catch (error) {
      dispatch({ type: 'ADD_POST_FAILURE', payload: error.message });
    }
  };

  const updatePost = async (updatedPost) => {
    dispatch({ type: 'UPDATE_POST_REQUEST' });
    try {
      const response = await axios.put(`/api/posts/${updatedPost.id}`, updatedPost); // Replace with your API endpoint
      dispatch({ type: 'UPDATE_POST_SUCCESS', payload: response.data });
    } catch (error) {
      dispatch({ type: 'UPDATE_POST_FAILURE', payload: error.message });
    }
  };

  return (
    <PostContext.Provider value={{ state, fetchPosts, addPost, updatePost }}>
      {children}
    </PostContext.Provider>
  );
};

export const usePosts = () => {
  return useContext(PostContext);
};

2. Usage in a React Component (e.g., App.js)

import React, { useEffect } from 'react';
import { PostProvider, usePosts } from './PostContext';

const PostList = () => {
  const { state, fetchPosts, addPost, updatePost } = usePosts();
  const { posts, loading, error } = state;

  useEffect(() => {
    fetchPosts();
  }, []);

  const handleAddPost = () => {
    const newPost = { title: 'New Post', body: 'This is a new post.' };
    addPost(newPost);
  };

  const handleUpdatePost = (post) => {
    const updatedPost = { ...post, title: 'Updated Title' };
    updatePost(updatedPost);
  };

  if (loading) return <p>Loading posts...</p>;
  if (error) return <p>Error: {error}</p>;

  return (
    <div>
      <h1>Posts</h1>
      <button onClick={handleAddPost}>Add New Post</button>
      <ul>
        {posts.map((post) => (
          <li key={post.id}>
            {post.title} - {post.body}
            <button onClick={() => handleUpdatePost(post)}>Update</button>
          </li>
        ))}
      </ul>
    </div>
  );
};

function App() {
  return (
    <PostProvider>
      <PostList />
    </PostProvider>
  );
}

export default App;
